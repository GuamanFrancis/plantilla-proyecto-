// utils/storage.ts
const TOKEN_KEY = 'auth_token'

export function saveToken(token: string): void {
  localStorage.setItem(TOKEN_KEY, token)
}

export function getToken(): string | null {
  return localStorage.getItem(TOKEN_KEY)
}

export function removeToken(): void {
  localStorage.removeItem(TOKEN_KEY)
}

ARCHIVO: src/api/client.ts
POR QUÉ: Configura Axios una sola vez para que:
  1. Sepa la URL del backend (no la repetimos en cada llamada)
  2. Adjunte el token automáticamente en cada request (interceptor)
  3. Si el backend responde 401, borre el token y mande al login (interceptor)
Solo cambias el puerto (3000) si tu backend usa otro.

──────────────────────────────
// api/client.ts
import axios from 'axios'
import { getToken, removeToken } from '../utils/storage'

const client = axios.create({
  baseURL: 'http://localhost:3000/api',  // ← CAMBIA EL PUERTO SI ES NECESARIO
  headers: {
    'Content-Type': 'application/json',
  },
})

// Antes de enviar cada request: adjunta el token si existe
client.interceptors.request.use((config) => {
  const token = getToken()
  if (token) {
    config.headers.Authorization = `Bearer ${token}`
  }
  return config
})

// Cuando llega la respuesta: si es 401, sesión expiró
client.interceptors.response.use(
  (response) => response,
  (error) => {
    if (error.response?.status === 401) {
      removeToken()
      window.location.href = '/login'
    }
    return Promise.reject(error)
  }
)

export default client

// context/AuthContext.tsx
import { createContext, useState, useEffect } from 'react'
import type { ReactNode } from 'react'
import type { User } from '../types/entities'
import { getToken, saveToken, removeToken } from '../utils/storage'
import client from '../api/client'
import type { AuthContextType } from '../types/AuthContext'

export const AuthContext = createContext<AuthContextType | null>(null)

export function AuthProvider({ children }: { children: ReactNode }) {
  const [user, setUser] = useState<User | null>(null)
  const [token, setToken] = useState<string | null>(getToken())
  const [isLoading, setIsLoading] = useState(true)

  // Al iniciar la app: si había token guardado, verifica que siga siendo válido
  useEffect(() => {
    const storedToken = getToken()
    if (!storedToken) {
      setIsLoading(false)
      return
    }
    client.get('/auth/me')
      .then((res) => {
        setUser(res.data.data.user)
        setToken(storedToken)
      })
      .catch(() => {
        removeToken()
        setUser(null)
        setToken(null)
      })
      .finally(() => setIsLoading(false))
  }, [])

  function login(newToken: string, newUser: User) {
    saveToken(newToken)
    setToken(newToken)
    setUser(newUser)
  }

  function logout() {
    removeToken()
    setToken(null)
    setUser(null)
  }

  return (
    <AuthContext.Provider value={{ user, token, isLoading, login, logout }}>
      {children}
    </AuthContext.Provider>
  )
}

// routes/PrivateRoute.tsx
import { Navigate, Outlet } from 'react-router-dom'
import { useAuth } from '../hooks/useAuth'

function PrivateRoute() {
  const { token, isLoading } = useAuth()

  if (isLoading) return <div>Cargando...</div>

  return token ? <Outlet /> : <Navigate to="/login" replace />
}

export default PrivateRoute

Estructura CLAVE de pacientes.api.ts:
  export interface CreatePacienteDto {
    nombre: string; apellido: string; cedula: string
    telefono: string; email?: string
  }
  export type UpdatePacienteDto = Partial<CreatePacienteDto>

  export async function getPacientes(): Promise<Paciente[]> {
    const res = await client.get<ApiResponse<Paciente[]>>('/pacientes')
    return res.data.data
  }
  // createPaciente, updatePaciente, deletePaciente siguen mismo patrón

Patrón clave en PacientesPage.tsx:
  const EMPTY_FORM: CreatePacienteDto = { nombre:'', apellido:'', cedula:'', ... }
  const [formData, setFormData] = useState(EMPTY_FORM)
  const [editingId, setEditingId] = useState<number | null>(null)

  function handleChange(e: React.ChangeEvent<HTMLInputElement>) {
    const { name, value } = e.target
    setFormData(prev => ({ ...prev, [name]: value }))  // ← patrón reutilizable
  }

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault()
    if (editingId !== null) await updatePaciente(editingId, formData)
    else await createPaciente(formData)
    handleCancel(); await loadPacientes()
  }

  function handleEdit(p: Paciente) {
    setFormData({ nombre: p.nombre, ... }); setEditingId(p.id)
    window.scrollTo({ top: 0, behavior: 'smooth' })
  }

  async function handleDelete(id: number) {
    if (!confirm('¿Eliminar?')) return
    await deletePaciente(id); await loadPacientes()
  }
