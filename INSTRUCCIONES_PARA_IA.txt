PATRONES IMPORTANTES DEL BACKEND (info clave para las páginas CRUD)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
TODAS las respuestas del backend tienen esta forma:
  Éxito:  { success: true,  data: <objeto o array> }
  Error:  { success: false, error: { code, message, details? } }

Axios guarda la respuesta HTTP en .data, así que para llegar a los datos:
  const res = await client.get('/pacientes')
  const pacientes = res.data.data  // ← DOS .data (uno de axios, otro del backend)

ENDPOINTS POR MÓDULO:
  Pacientes:
    GET    /api/pacientes       → lista con { success: true, data: Paciente[] }
    GET    /api/pacientes/:id   → uno con { success: true, data: Paciente }
    POST   /api/pacientes       → body { nombre, apellido, cedula, telefono, email? }
    PUT    /api/pacientes/:id   → body con campos opcionales
    DELETE /api/pacientes/:id   → { success: true, data: { message: "..." } }

  Especialidades: mismo patrón, campos { nombre, descripcion? }

  Citas:
    GET    /api/citas           → lista con paciente y especialidad incluidos
    GET    /api/citas/:id
    POST   /api/citas           → body { pacienteId, especialidadId, fecha, motivo? }
    DELETE /api/citas/:id       → NO TIENE PUT/UPDATE
